// imports without aliasing
#import "Basic";
#import "GL";
#import "Math";
#import "Window_Creation";

// imports with aliasing
Input :: #import "Input";
Simp  :: #import "Simp";
Windows :: #import "Windows";

// comment

main :: () {
    run := true;
    player_pos := Vector2.{40, 40};

    prev_time := seconds_since_init();

    while run {
        cur_time := seconds_since_init();
        dt := cast(float)(cur_time - prev_time);
        prev_time = cur_time;

        for Input.events_this_frame {
            if it.type == .QUIT {
                run = false;
            }

            SPEED :: 200;

            if it.type == {
                case .KEYBOARD;  // very weird to use ; here.
                    if it.key_pressed && it.key_code == #char "A" {  // so much indentation.
                        player_pos.x -= SPEED * dt;
                    }
                    if it.key_pressed && it.key_code == #char "D" {
                        player_pos.x += SPEED * dt;
                    }
            }
        }
    }
}

  • (2025-08-04)

  • Closed beta.

  • Bad aspects of syntax :

    • The need for ;  is a bit annoying, maybe.

    • #char  looks ugly, but maybe I didn‚Äôt understand its purpose correctly.

    • I don‚Äôt like pos := Vector2.{40, 40}; , the need for .  is strange.

    • The way pointers are used is odd, doing deref with something.* .

  • Okay aspects of syntax :

    • I don‚Äôt fully understand how it works, but the existence of default values for structs and enums is nice.

    • Aliases in imports are not mandatory.

    • Strange libraries that don‚Äôt expose the core of how things work.

      • "Input" for inputs.

      • Etc.

    • Still, it feels redundant. There are already libraries that do exactly what these do, like GLFW, etc.

    • Doesn‚Äôt seem to solve any problem.

  • I don‚Äôt have many opinions on meta-language.
